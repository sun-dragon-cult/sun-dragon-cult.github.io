This is an incomplete setting for Item Piles. What's missing is defining what items are unique.
Running this in the terminal or through a macro will set the settings in Item Piles.

```js
game.itempiles.API.setActorClassType("character");
game.itempiles.API.setItemQuantityAttribute("system.quantity");
game.itempiles.API.setItemPriceAttribute("system.price.real");
game.itempiles.API.setItemFilters([
  {
    path: "type",
    filters: [
      "cult",
      "hitLocation",
      "homeland",
      "occupation",
      "passion",
      "rune",
      "runeMagic",
      "skill",
      "spiritMagic",
    ].join(","),
  },
  {
    path: "system.isNatural",
    filters: "true",
  },
]);
game.itempiles.API.setCurrencies([
  {
    type: "gear",
    name: "Wheels",
    primary: false,
    img: "systems/rqg/assets/images/gear/wheel.webp",
    abbreviation: "{#}W",
    exchangeRate: 10,
    data: {
      item: {
        name: "Wheels",
        type: "gear",
        system: {
          physicalItemType: "currency",
          price: {
            real: 10,
          },
        },
      },
    },
  },
  {
    type: "gear",
    name: "Lunars",
    primary: true,
    img: "systems/rqg/assets/images/gear/lunar.webp",
    abbreviation: "{#}L",
    exchangeRate: 1,
    data: {
      item: {
        name: "Lunars",
        type: "gear",
        system: {
          physicalItemType: "currency",
          price: {
            real: 1,
          },
        },
      },
    },
  },
  {
    type: "gear",
    name: "Clacks",
    primary: false,
    img: "systems/rqg/assets/images/gear/clack.webp",
    abbreviation: "{#}C",
    exchangeRate: 0.1,
    data: {
      item: {
        name: "Clacks",
        type: "gear",
        system: {
          physicalItemType: "currency",
          price: {
            real: 0.1,
          },
        },
      },
    },
  },
  {
    type: "gear",
```
